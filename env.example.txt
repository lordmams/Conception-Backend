# ===================================
# Configuration Serveur
# ===================================

# Port d'écoute de l'API
PORT=3000

# Environnement (development, production, test)
NODE_ENV=development

# ===================================
# Configuration MongoDB (NoSQL)
# ===================================

# URI de connexion MongoDB
# Format: mongodb://[username:password@]host[:port]/database
MONGODB_URI=mongodb://localhost:27017/gamedb

# En production avec authentification:
# MONGODB_URI=mongodb://admin:password@mongodb:27017/gamedb?authSource=admin

# ===================================
# Configuration JWT
# ===================================

# Clé secrète pour signer les tokens JWT
# ⚠️ IMPORTANT: Changez cette valeur en production !
# Générez une clé aléatoire avec: openssl rand -base64 32
JWT_SECRET=votre_clé_secrète_super_sécurisée_changez_moi_en_production

# Durée de validité des tokens JWT
# Format: 1h, 24h, 7d, 30d
JWT_EXPIRES_IN=24h

# ===================================
# Configuration CORS
# ===================================

# Origines autorisées pour CORS
# En développement: * (toutes les origines)
# En production: liste des domaines autorisés séparés par des virgules
CORS_ORIGIN=*

# En production:
# CORS_ORIGIN=https://monapp.com,https://www.monapp.com

# ===================================
# Configuration Sauvegardes MongoDB
# ===================================

# Hôte MongoDB pour les sauvegardes
MONGO_HOST=mongodb

# Port MongoDB
MONGO_PORT=27017

# Nom de la base de données
MONGO_DB=gamedb

# Répertoire des sauvegardes
BACKUP_DIR=/backups

# Nombre de jours de rétention des sauvegardes
RETENTION_DAYS=7

# Fuseau horaire
TZ=Europe/Paris

# Sauvegarder au démarrage du conteneur (true/false)
BACKUP_ON_START=false

# ===================================
# Configuration PostgreSQL (SQL) - Optionnel
# ===================================

# Hôte PostgreSQL
POSTGRES_HOST=localhost

# Port PostgreSQL
POSTGRES_PORT=5432

# Nom de la base de données PostgreSQL
POSTGRES_DB=gamedb_sql

# Utilisateur PostgreSQL
POSTGRES_USER=postgres

# Mot de passe PostgreSQL
POSTGRES_PASSWORD=postgres

# URL de connexion PostgreSQL complète (alternative)
# DATABASE_URL=postgresql://postgres:postgres@localhost:5432/gamedb_sql

# ===================================
# Configuration Rate Limiting
# ===================================

# Limite générale (requêtes par fenêtre)
RATE_LIMIT_GENERAL=100

# Fenêtre de temps pour rate limiting général (en minutes)
RATE_LIMIT_WINDOW_GENERAL=15

# Limite pour l'authentification (tentatives par fenêtre)
RATE_LIMIT_AUTH=5

# Fenêtre de temps pour auth rate limiting (en minutes)
RATE_LIMIT_WINDOW_AUTH=15

# ===================================
# Configuration Logging
# ===================================

# Niveau de log (debug, info, warn, error)
LOG_LEVEL=info

# ===================================
# Configuration Tests
# ===================================

# URI MongoDB pour les tests
MONGODB_TEST_URI=mongodb://localhost:27017/gamedb_test
